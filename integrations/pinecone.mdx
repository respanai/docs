## What is Pinecone?

[Pinecone](https://www.pinecone.io/) is a managed vector database for building search and retrieval systems. It handles indexing, storage, and similarity search at scale.

<Step title="Initialize Respan and query Pinecone">

Respan auto-instruments Pinecone — upsert, query, and delete operations are captured as spans.

<CodeGroup>
```python Python
from respan_tracing import RespanTelemetry
from respan_tracing.decorators import workflow, task
from pinecone import Pinecone
from openai import OpenAI

# Initialize — auto-instruments Pinecone
telemetry = RespanTelemetry()
client = OpenAI()
pc = Pinecone()
index = pc.Index("my-index")


def get_embedding(text: str) -> list[float]:
    response = client.embeddings.create(model="text-embedding-3-small", input=text)
    return response.data[0].embedding


@task(name="search_docs")
def search_docs(query: str):
    query_vector = get_embedding(query)
    results = index.query(vector=query_vector, top_k=3, include_metadata=True)
    return [match.metadata["text"] for match in results.matches]


@workflow(name="rag_pipeline")
def rag_pipeline(query: str):
    context = search_docs(query)
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": f"Context: {chr(10).join(context)}"},
            {"role": "user", "content": query},
        ],
    )
    return response.choices[0].message.content


result = rag_pipeline("How does tracing work?")
print(result)
```
```typescript JavaScript
import { RespanTelemetry } from '@respan/tracing';
import { Pinecone } from '@pinecone-database/pinecone';
import OpenAI from 'openai';

const respan = new RespanTelemetry({
    apiKey: process.env.RESPAN_API_KEY || "",
    appName: 'pinecone-app',
    instrumentModules: { pinecone: Pinecone },
});
await respan.initialize();

const pc = new Pinecone();
const index = pc.index('my-index');
const openai = new OpenAI();

async function getEmbedding(text: string) {
    const response = await openai.embeddings.create({
        model: 'text-embedding-3-small',
        input: text,
    });
    return response.data[0].embedding;
}

// Query — auto-traced
async function ragPipeline(query: string) {
    return await respan.withWorkflow({ name: 'rag_pipeline' }, async () => {
        const context = await respan.withTask({ name: 'search_docs' }, async () => {
            const queryVector = await getEmbedding(query);
            const results = await index.query({
                vector: queryVector,
                topK: 3,
                includeMetadata: true,
            });
            return results.matches?.map(m => m.metadata?.text as string) || [];
        });

        const completion = await openai.chat.completions.create({
            model: 'gpt-4o-mini',
            messages: [
                { role: 'system', content: `Context: ${context.join('\n')}` },
                { role: 'user', content: query },
            ],
        });
        return completion.choices[0].message.content;
    });
}

ragPipeline("How does tracing work?").then(console.log);
```

## Configuration

Pinecone is auto-instrumented via `Instruments.PINECONE` (Python) or `pinecone` (JavaScript). No additional configuration is needed.

See the [Python Tracing SDK reference](/sdks/python/tracing/main/initialize) or [JavaScript Tracing SDK reference](/sdks/typescript/tracing/main/initialize) for configuration options.
